#include <bits/stdc++.h>
using namespace std;

const int N = 1e5;
int arr[N];
long long seg[4*N];

// Build segment tree
void build(int idx, int low, int high) {
    if (low == high) {
        seg[idx] = arr[low];
        return;
    }
    int mid = (low + high) / 2;
    build(2*idx+1, low, mid);
    build(2*idx+2, mid+1, high);
    seg[idx] = seg[2*idx+1] + seg[2*idx+2];
}

// Point update (update one index)
void update(int idx, int low, int high, int pos, int val) {
    if (low == high) {
        seg[idx] = val;
        return;
    }
    int mid = (low + high) / 2;
    if (pos <= mid) update(2*idx+1, low, mid, pos, val);
    else update(2*idx+2, mid+1, high, pos, val);

    seg[idx] = seg[2*idx+1] + seg[2*idx+2];
}

// Range update (no lazy)
void updateRange(int idx, int low, int high, int l, int r, int val) {
    // no overlap
    if (high < l || low > r) return;

    // leaf node
    if (low == high) {
        seg[idx] += val;
        return;
    }

    int mid = (low + high) / 2;
    updateRange(2*idx+1, low, mid, l, r, val);
    updateRange(2*idx+2, mid+1, high, l, r, val);
    seg[idx] = seg[2*idx+1] + seg[2*idx+2];
}

// Range sum query
long long query(int idx, int low, int high, int l, int r) {
    if (high < l || low > r) return 0; // no overlap
    if (low >= l && high <= r) return seg[idx]; // total overlap

    int mid = (low + high) / 2;
    return query(2*idx+1, low, mid, l, r)
         + query(2*idx+2, mid+1, high, l, r);
}

int main() {
    int n = 6;
    int a[] = {1, 3, 5, 7, 9, 11};
    for (int i = 0; i < n; i++) arr[i] = a[i];

    build(0, 0, n-1);

    cout << "Initial sum [1,3]: " << query(0, 0, n-1, 1, 3) << endl;

    updateRange(0, 0, n-1, 1, 3, 10); // add 10 to each element in [1,3]

    cout << "After update sum [1,3]: " << query(0, 0, n-1, 1, 3) << endl;
}
